
\chapter{Division}


\section{Introducing Divides}


\highlightdef{ $b | a$ iff $b mod a \equiv 0$ }

\begin{example}
True of false: \\
\textbf{(a)} $b$ must be less than $a$ for $b | a$ to hold. 
\end{example}


\highlightdef{ $b | (x_1 x_2 \cdots x_n + y_1 y_2 \cdots y_n)$ iff $b | (x_1 x_2 \cdots x_n)$ 
and $b | (x_1 x_2 \cdots x_n)$ }




\highlightdef{ $b | (x_1 x_2 \cdots x_n + 1)$ iff $b = \pm 1$ }





\frmrule

\begin{example}
By using the division relation, show that the square of any integer 
is of the form $4k$ or $4k+1$. Hence, or otherwise show that 
none of the numbers in the sequence 11, 111, 1111, 11111, can be square.
\end{example}
